```typescript
  abbreviation: string;
  name: string;
  hasSalesTax: boolean;
  stateRate: string;
  stateRateNum: number;
  combinedAvgRate: string;
  combinedRateRange: string;
  taxType: string;
  nexus: {
    revenueThreshold: string | null;
    transactionThreshold: string | null;
    bothRequired: boolean;
    effectiveDate: string;
    note: string;
  } | null;
  exemptions: {
    groceries: string;

    prescriptionNote: string;
  };
  hasLocalTax: boolean;
  localTaxRange: string;
  filingFrequency: string;
  filingDeadline: string;
  officialWebsite: string;
  salesTaxPage: string;
  registrationUrl: string;
  contactPhone: string;
  [key: string]: unknown;
}

// Keep backward compat alias
export type StateData = StateEntry;

// TaxabilityRule — per-category tax treatment for a state
export interface TaxabilityRule {
  status: "taxable" | "exempt" | "reduced";
  rate?: number;
  threshold?: number;
  thresholdType?: "exempt_below" | "tax_excess";
  note: string;
}

// ZipEntry — one tax jurisdiction within a ZIP code
export interface ZipEntry {
  region: string;
  combined: number;
  county: number;
  city: number;
  special: number;
}

// StateTaxData — per-state JSON file produced by build pipeline
export interface StateTaxData {
  state: string;
  stateName: string;
  stateRate: number;
  hasSalesTax: boolean;
  updated: string;
  taxability: {
    groceries: TaxabilityRule;
    clothing: TaxabilityRule;
    medicine: TaxabilityRule;
    candy: TaxabilityRule;
  };
  zips: Record<string, ZipEntry[]>;
}

// TaxDataManifest — manifest.json produced by build pipeline
export interface TaxDataManifest {
  generated: string;
  ratesAsOf: string;
  stateCount: number;
  totalZipCodes: number;
  states: string[];
}

// TaxLineItem and TaxSummary — used by calculator
export interface TaxLineItem {
  product: import("./products").Product;
  quantity: number;
  subtotal: number;
  taxableAmount: number;
  taxRate: number;
  taxAmount: number;
  explanation: string;
  isExempt: boolean;
}

export interface TaxSummary {
  state: StateEntry;
  items: TaxLineItem[];
  subtotal: number;
  totalTax: number;
  total: number;
  effectiveRate: number;
  notes: string[];
}
```

**Create `scripts/build-tax-data.mjs`** (per D-06, D-07, D-08) — the Avalara CSV-to-JSON pipeline.

The script must:

1. **Read CSV files** from `data/avalara/` directory. Each CSV has columns: `State,ZipCode,TaxRegionName,EstimatedCombinedRate,StateRate,EstimatedCountyRate,EstimatedCityRate,SpecialRate,RiskLevel`. Use simple `String.split(",")` parsing (no library needed per research). Pad ZIP codes to 5 digits with `.padStart(5, "0")` to preserve leading zeros.

2. **Read taxability rules** from `data/taxability.json`. For each state, resolve the taxability rule: if the state has an override entry, use it; otherwise use the category default. Construct a `TaxabilityRule` object for each of the 4 categories (groceries, clothing, medicine, candy).

3. **Normalize data** by grouping CSV rows: first by state, then by ZIP code. Each ZIP code maps to an array of `ZipEntry` objects (one per TaxRegionName — handles multi-jurisdiction ZIPs).

4. **Validate** (per D-08). The pipeline MUST fail with a non-zero exit code and descriptive error if any of these checks fail:
   - Missing states: fewer than 46 states with data (45 taxable + DC; no-tax states DE/MT/NH/OR may or may not have CSV files)
   - Rate anomalies: any rate < 0 or any combined rate > 0.15
   - Duplicate ZIPs: same ZIP + same region name appearing more than once within a state
   - State rate consistency: if the state rate differs across ZIPs within the same state file (log warning; some states have minor variation)

5. **Generate no-tax state files**: For DE, MT, NH, OR (and AK which has no state tax but local taxes), create JSON files with `hasSalesTax: false`, `stateRate: 0`, empty zips object, and all taxability rules set to `{ "status": "exempt", "note": "No state sales tax" }`. If an Avalara CSV exists for these states (AK has local taxes), still parse it but set `hasSalesTax: false` and `stateRate: 0`.

6. **Write per-state JSON** to `public/tax-data/{STATE}.json` (e.g., `public/tax-data/CA.json`). Upper-case two-letter abbreviation. Each file follows the `StateTaxData` interface schema. Create the `public/tax-data/` directory if it does not exist.

7. **Write manifest** to `public/tax-data/manifest.json` following the `TaxDataManifest` interface. The `ratesAsOf` field should be derived from the CSV filenames (e.g., `TAXRATES_ZIP5_AL202603.csv` -> "March 2026") or fall back to the current build month/year. The `generated` field is the current ISO timestamp.

8. **Console output**: Print summary: number of states processed, total ZIP codes, any warnings.

Use the same ESM patterns as the existing `build-data.mjs`: `import { readFile, writeFile, readdir, mkdir } from "node:fs/promises"`, `fileURLToPath`, etc.

Include a `STATE_NAMES` lookup object (copy from `build-data.mjs` or import the same data) for populating `stateName` in the output.

The no-tax states list: `["DE", "MT", "NH", "OR"]`. Alaska (`AK`) has no state sales tax but has local taxes, so it is handled specially: if an AK CSV exists, parse it normally but set `stateRate: 0` and `hasSalesTax: false`.
